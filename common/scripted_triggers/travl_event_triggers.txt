travl_event_is_location_terrain_expert_trigger = {
	OR = {
		AND = {
			location = {
				OR = {
					terrain = forest
					terrain = taiga
				}
			}
			has_trait = forest_fighter
		}
		AND = {
			location = {
				OR = {
					terrain = farmlands
					terrain = plains
					terrain = steppe
				}
			}
			has_trait = open_terrain_expert
		}
		AND = {
			location = {
				OR = {
					terrain = hills
					terrain = mountains
					terrain = wetlands
				}
			}
			has_trait = rough_terrain_expert
		}
		AND = {
			location = {
				OR = {
					terrain = desert
					terrain = desert_mountains
					terrain = drylands
					terrain = oasis
				}
			}
			has_trait = desert_warrior
		}
		AND = {
			location = {
				terrain = jungle
			}
			has_trait = jungle_stalker
		}
		AND = {
			location = {
				OR = {
					terrain = sea
					terrain = coastal_sea
				}
			}
			has_trait = forder
		}
	}
}

travl_is_hostile_character_trigger = {
	save_temporary_scope_as = character
	OR = {
		is_at_war_with = $CHARACTER$
		any_hostile_raider = { 
			this = $CHARACTER$
		}
		$CHARACTER$ = {
			any_hostile_raider = {
				this = scope:character
			}
		}
	}
}

travl_is_travel_danger_type_mountainous_terrain_trigger = {
	OR = {
		travel_danger_type = {
			travel_plan = $TRAVEL_PLAN$
			terrain = mountains
		}
		travel_danger_type = {
			travel_plan = $TRAVEL_PLAN$
			terrain = desert_mountains
		}
	}
}
