# Invite guests.
# Human player only.
travl_invite_guests_interaction = {
	category = interaction_category_vassal
	desc = travl_invite_guests_interaction_desc
	icon = guest

	populate_recipient_list = {
		every_living_character = {
			limit = {
				is_ai = yes
				in_diplomatic_range = scope:actor
				trigger_if = {
					limit = {
						NOT = { has_game_rule = travl_interaction_range_all }
					}
					location ?= {
						squared_distance = {
							target = scope:actor.capital_province
							value <= scope:actor.travl_interaction_range_value
						}
					}
				}
				save_temporary_scope_as = invitee
				scope:actor = {
					is_character_interaction_potentially_accepted = {
						interaction = travl_invite_as_guest_interaction
						recipient = scope:invitee
					}
				}
			}
			add_to_list = characters
		}
	}

	is_shown = {
		scope:actor = scope:recipient
	}

	is_valid_showing_failures_only = {
		scope:actor = {
			is_busy_in_events_localised = yes
			is_imprisoned = no
			custom_tooltip = {
				text = travl_invite_as_guest_actor_cooldown_tt
				NOT = { has_variable = travl_invite_as_guest_actor_cooldown }
			}
		}
	}

	on_accept = {
		show_as_tooltip = {
			scope:secondary_recipient = {
				every_traveling_family_member = {
					scope:actor = {
						add_visiting_courtier = prev
					}
				}
			}
		}

		open_interaction_window = {
			interaction = travl_invite_as_guest_interaction
			actor = scope:actor
			recipient = scope:secondary_recipient
		}
	}

	auto_accept = yes
}
