# Invite courtiers.
# Scopes: actor / recipient is the buyer, secondary_recipient is character being invited.
# Human player only.
travl_invite_courtiers_interaction = {
	category = interaction_category_vassal
	desc = travl_invite_courtiers_interaction_desc
	icon = guest

	populate_recipient_list = {
		every_living_character = {
			limit = {
				is_ai = yes
				in_diplomatic_range = scope:actor
				save_temporary_scope_as = recruitee
				scope:actor = {
					is_character_interaction_potentially_accepted = {
						interaction = invite_to_court_interaction
						recipient = scope:recruitee
					}
				}
			}
			add_to_list = characters
		}
	}

	is_shown = {
		scope:actor = scope:recipient
	}

	is_valid_showing_failures_only = {
		scope:actor = {
			is_busy_in_events_localised = yes
			is_imprisoned = no
		}
	}

	on_accept = {
		show_as_tooltip = {
			scope:secondary_recipient = {
				every_traveling_family_member = {
					limit = {
						is_available = yes
						is_at_location = scope:secondary_recipient.location
					}
					scope:actor = { add_courtier = prev }
				}
			}

			scope:actor = {
				stress_impact = {
					shy = minor_stress_impact_gain
				}
			}
		}

		open_interaction_window = {
			interaction = invite_to_court_interaction
			actor = scope:actor
			recipient = scope:secondary_recipient
		}
	}

	auto_accept = yes
}
